        document.getElementById("searchBox").addEventListener("keyup", function() {
            let searchText = this.value.toLowerCase();
            let books = document.querySelectorAll(".book-item");

            books.forEach(book => {
                let title = book.querySelector(".book-title").textContent.toLowerCase();
                let author = book.querySelector(".book-author").textContent.toLowerCase();
                let category = book.querySelector(".book-category").textContent.toLowerCase();

                if (title.includes(searchText) || author.includes(searchText) || category.includes(searchText)) {
                    book.style.display = "table-row"; // Show matching books
                } else {
                    book.style.display = "none"; // Hide non-matching books
                }
            });
        });
    </script>

5. server.js

const express = require('express');
const mongoose = require('./config/db'); // MongoDB connection
const bodyParser = require('body-parser');
const methodOverride = require('method-override');

const app = express();

// Middleware
app.use(bodyParser.urlencoded({ extended: true }));
app.use(methodOverride('_method'));
app.use(express.static('public'));
app.set('view engine', 'ejs');

// Routes
const bookRoutes = require('./routes/bookRoutes');
app.use('/books', bookRoutes);

// Home Route
app.get('/', (req, res) => {
    res.render('index');
});

// Start Server
const PORT = 3000;
app.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`));
